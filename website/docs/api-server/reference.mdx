import BrowserOnly from "@docusaurus/BrowserOnly";
import useBaseUrl from '@docusaurus/useBaseUrl';
import "swagger-ui-react/swagger-ui.css";


export const APIUI = ({ url }) => (
  // FIXME: implement dark mode for swagger-ui somehow.
  // Adds a white background to the Swagger UI
  // So you can still use it in dark mode.
  <div
    style={{
      backgroundColor: "#fff",
      borderRadius: "4px",
      padding: "2px 0",
      margin: "0 -20px",
    }}
  >
    <BrowserOnly fallback={<div>Loading...</div>}>
      {() => {
        const SwaggerUI = require("swagger-ui-react").default;
        const swaggerUrl = useBaseUrl(url);
        return (
          <SwaggerUI
            url={swaggerUrl}
            // Turn on hash based deeplinking so that each API call can be linked to directly.
            // Seems to work okay w/ docusaurus' routing.
            deepLinking
            // turn off the "try this out" stuff.
            supportedSubmitMethods={[]}
            // Collapse all models sections by default, they're very verbose.
            defaultModelsExpandDepth={0}
          />
        );
      }}
    </BrowserOnly>
  </div>
);
